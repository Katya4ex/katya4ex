def expos_distance(dis):
    if dis < pd.Timedelta('180 days'):
        return ' 6 '
    elif dis < pd.Timedelta('365 days') and dis >= pd.Timedelta('180 days'):
        return ' 6   1 '
    elif dis >= pd.Timedelta('365 days'):
        return ' 1 '
